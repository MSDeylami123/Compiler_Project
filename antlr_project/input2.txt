int compute(int limit) {
    int total = 0;
    int index = 0;

    while (index <= limit) {

        if (index / 2 == 0) {
            index = index + 1;
        } else {
            total = total + index;
            index = index + 1;
        }
    }

    return total;
}

int start() {

    int value = compute(10);

    if (value <= 20) {
        value = value + 5;
    } else {
        value = value - 5;
    }

    return value;
}
